1//! Logger utilities configuráveis via *feature flags*.
2//!
3//! Este arquivo define reexports de macros de logging (`info!`, `trace!`,
4//! `warn!`, `debug!`, `error!`) e funções utilitárias para inicialização do
5//! backend de logging apropriado conforme as *features* ativadas.
6//!
7//! Suporta atualmente as features: `log-itm`, `defmt`, `log-semihost`.
8//!
9//! A escolha do backend altera quais crates auxiliares são utilizados
10//! (ex.: `cortex-m-log`, `defmt-rtt`, `cortex-m-semihosting`).
11
12#![allow(unsafe_code)]
13
14// Primeiro, reexporta as macros/funções de logging dependendo da feature.
15cfg_if::cfg_if! {
16    // Se a feature `defmt` estiver habilitada, reexportamos os símbolos do
17    // `defmt` (que fornece um logger eficiente para embarcados).
18    if #[cfg(feature = "defmt")] {
19        #[allow(unused_imports)]
20        pub use defmt::{info, trace, warn, debug, error};
21
22    // Caso contrário, reexportamos a API compatível com `log`.
23    } else {
24        #[allow(unused_imports)]
25        pub use log::{info, trace, warn, debug, error};
26    }
27}
28
29// Em seguida, escolhemos o backend concreto para `init()` usando outra
30// condição `cfg_if!` — cada branch configura o logger conforme o destino.
31cfg_if::cfg_if! {
32    // -- Backend: ITM (via `cortex-m-log`) -------------------------------------------------
33    if #[cfg(feature = "log-itm")] {
34        // Em builds normais, usamos `panic_itm` para saída de panic via ITM.
35        #[cfg(not(test))]
36        use panic_itm as _;
37
38        // lazy_static para instanciar o LOGGER globalmente.
39        use lazy_static::lazy_static;
40        use log::LevelFilter;
41
42        // Reexporta o tipo `Logger` para consumidores que queiram usá-lo.
43        pub use cortex_m_log::log::Logger;
44
45        // Importa tipos auxiliares do `cortex_m_log` para construir o LOGGER.
46        use cortex_m_log::{
47            destination::Itm as ItmDest,
48            printer::itm::InterruptSync,
49            modes::InterruptFree,
50            printer::itm::ItmSync
51        };
52
53        // Instância estática do LOGGER configurada para ITM.
54        lazy_static! {
55            pub static ref LOGGER: Logger<ItmSync<InterruptFree>> = Logger {
56                level: LevelFilter::Info,
57                inner: unsafe {
58                    // Criamos um destino ITM sem usar `Peripherals::steal()`.
59                    InterruptSync::new(
60                        // Transmute é usado para construir o tipo corretamente.
61                        ItmDest::new(core::mem::transmute::<(), stm32g4xx_hal::stm32::ITM>(()))
62                    )
63                },
64            };
65        }
66
67        // Macro `println!` local que delega para `log::info!()`.
68        #[allow(unused_macros)]
69        macro_rules! println {
70            ($($arg:tt)+) => {
71                log::info!($($arg)+);
72            };
73        }
74
75        // Expõe a macro `println` para uso dentro do crate.
76        #[allow(unused_imports)]
77        pub(crate) use println;
78
79        // Inicializa o logger global baseado em `LOGGER`.
80        #[allow(dead_code)]
81        pub fn init() {
82            cortex_m_log::log::init(&LOGGER).unwrap();
83        }
84
85    // -- Backend: defmt (via RTT) ---------------------------------------------------------
86    }
87    else if #[cfg(feature = "defmt")] {
88        // `defmt_rtt` fornece backend RTT para `defmt`.
89        use defmt_rtt as _; // global logger
90        #[cfg(not(test))]
91        use panic_probe as _;
92
93        #[allow(unused_imports)]
94        pub use defmt::Logger;
95        #[allow(unused_imports)]
96        pub use defmt::println;
97
98        // `defmt` não precisa de inicialização explícita aqui.
99        #[allow(dead_code)]
100        pub fn init() {}
101
102    // -- Backend: semihosting (via `cortex-m-log`) --------------------------------------
103    }
104    else if #[cfg(feature = "log-semihost")] {
105        #[cfg(not(test))]
106        use panic_semihosting as _;
107
108        use lazy_static::lazy_static;
109        use log::LevelFilter;
110
111        pub use cortex_m_log::log::Logger;
112        use cortex_m_log::printer::semihosting;
113        use cortex_m_log::printer::semihosting::Semihosting;
114        use cortex_m_log::modes::InterruptOk;
115        use cortex_m_semihosting::hio::HStdout;
116
117        // Logger que escreve via semihosting stdout.
118        lazy_static! {
119            static ref LOGGER: Logger<Semihosting<InterruptOk, HStdout>> = Logger {
120                level: LevelFilter::Info,
121                inner: semihosting::InterruptOk::<_>::stdout().expect("Get Semihosting stdout"),
122            };
123        }
124
125        // Macro `println!` que mapeia para hprintln! do semihosting.
126        #[allow(unused_macros)]
127        macro_rules! println {
128            ($s:expr) => {
129                cortex_m_semihosting::hprintln!($s).unwrap();
130            };
131            ($s:expr, $($tt:tt)*) => {
132                cortex_m_semihosting::hprintln!($s, $($tt)*).unwrap();
133            };
134        }
135
136        #[allow(unused_imports)]
137        pub(crate) use println;
138
139        #[allow(dead_code)]
140        pub fn init() {
141            cortex_m_log::log::init(&LOGGER).unwrap();
142        }
143
144    // -- Backend padrão: nenhum (fallback) --------------------------------------------
145    }
146    else {
147        // Se nenhum backend de logging foi selecionado, usamos `panic-halt` para
148        // definir o comportamento de panic sem imprimir nada.
149        #[cfg(not(test))]
150        use panic_halt as _;
151
152        // `init()` não precisa fazer nada no fallback.
153        #[allow(dead_code)]
154        pub fn init() {}
155    }
156}
