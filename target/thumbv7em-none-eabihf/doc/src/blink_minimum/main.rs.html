1//! Exemplo mínimo: piscar o LED da placa Nucleo G474RE.
2//!
3//! Este binário é construído com `no_std` e `cortex-m-rt` e demonstra um
4//! loop simples que alterna um pino GPIO para acender/apagar um LED.
5//!
6//! Comentários linha-a-linha foram adicionados para servir como documentação
7//! e facilitar entendimento durante leitura do código e geração de docs.
8
9// Não permitir warnings e código unsafe no exemplo — facilita ensino e testes.
10#![deny(warnings)]
11#![deny(unsafe_code)]
12// `no_main` : usamos o entry point fornecido pelo `cortex-m-rt`.
13#![no_main]
14// `no_std` : ambiente embarcado sem biblioteca padrão.
15#![no_std]
16
17// Importa traits de conveniência para configurar pinos e clocks.
18use hal::prelude::*;
19// Acesso às estruturas de periféricos dos pacotes HAL.
20use hal::stm32;
21// Alias do crate HAL para uso consistente no código.
22use stm32g4xx_hal as hal;
23
24// Macro `#[entry]` para marcar o ponto de entrada do programa.
25use cortex_m_rt::entry;
26
27// Importa o módulo `utils` que contém o logger e macros auxiliares.
28#[macro_use]
29mod utils;
30
31// Importa a macro `info!` reexportada pelo módulo `utils::logger`.
32use utils::logger::info;
33
34// Ponto de entrada da aplicação embarcada.
35#[entry]
36fn main() -> ! {
37    // Inicializa o logger (se configurado via features).
38    utils::logger::init();
39
40    // Mensagem de início (via logger configurado).
41    info!("start");
42
43    // Obtém acesso aos periféricos do microcontrolador.
44    // `take()` retorna `Some(Peripherals)` apenas uma vez; falha se já
45    // tiverem sido tomados por outra parte do código.
46    let dp = stm32::Peripherals::take().expect("cannot take peripherals");
47
48    // Constrói a configuração do Reset & Clock Control (RCC).
49    let mut rcc = dp.RCC.constrain();
50
51    // Log de inicialização do hardware usado neste exemplo.
52    info!("Init Led");
53
54    // Separa o controlador GPIOA em partes para configuração de pinos.
55    let gpioa = dp.GPIOA.split(&mut rcc);
56
57    // Configura PA5 como saída push-pull — pino do LED nas placas Nucleo.
58    let mut led = gpioa.pa5.into_push_pull_output();
59
60    // Loop principal infinito — alterna o estado do LED com delays simples.
61    loop {
62        // Registra que vamos colocar o LED em nível baixo (apagado/ligado
63        // dependendo do circuito da placa).
64        info!("Set Led low");
65
66        // Pequeno delay ativo: laço vazio para criar tempo visível.
67        for _ in 0..10_000_000 {
68            led.set_low().expect("deu errado o led low");
69        }
70
71        // Agora registamos que vamos colocar o LED em nível alto.
72        info!("Set Led High");
73
74        // Outro delay ativo para manter o LED no estado alto por um tempo.
75        for _ in 0..10_000_000 {
76            led.set_high().expect("deu errado o led low");
77        }
78    }
79}
